    gateway: # 网关配置

        locator: # 资源定位
          enabled: false # 取消默认路由配置，默认值就是false
      routes: # 定义静态路由
        - id: dept # 路由标记
          uri: http://provider-dept-8002:8002 # 配置路由地址
          predicates: # 路由谓词工厂
            - Path=/** # 匹配该地址下的全部子路径
